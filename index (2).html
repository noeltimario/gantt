VitClinic Patient Management System (PHP & SQL)

This block displays the complete set of files required for the CRUD (Create, Read, Update, Delete) application.

1️⃣ Database Creation (MySQL)

CREATE DATABASE vitclinic;

USE vitclinic;

CREATE TABLE `patients` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(100) NOT NULL,
  `species` VARCHAR(50) NOT NULL,
  `breed` VARCHAR(50) NOT NULL,
  `owner` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`id`)
);

2️⃣ config.php (Database Connection)

<?php
$host = "localhost";
$user = "root";
$pass = ""; // Change this password if needed
$db   = "vitclinic";

$conn = new mysqli($host, $user, $pass, $db);

if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}
?>

4️⃣ add.php (Add Patient)

<?php

if(isset($_POST['submit'])){

    $species = $_POST['species'];
    $breed = $_POST['breed'];

5️⃣ edit.php (Edit Patient)

<?php
include 'config.php';

$id = $_GET['id'];
$result = $conn->query("SELECT * FROM patients WHERE id=$id");
$row = $result->fetch_assoc();

if(isset($_POST['submit'])){
    // NOTE: In a real app, use prepared statements to prevent SQL injection!
    $name = $_POST['name'];
    $species = $_POST['species'];
    $breed = $_POST['breed'];
    $owner = $_POST['owner'];

    $conn->query("UPDATE patients SET name='$name', species='$species', breed='$breed', owner='$owner' WHERE id=$id");
    header("Location: index.php");
}
?>
<!DOCTYPE html>
<html>
<head>
    <title>Edit Patient</title>
</head>
<body>
    <h2>Edit Patient</h2>
    <form method="POST">
        Name: <input type="text" name="name" value="<?= $row['name'] ?>" required><br><br>
        Species: <input type="text" name="species" value="<?= $row['species'] ?>" required><br><br>
        Breed: <input type="text" name="breed" value="<?= $row['breed'] ?>" required><br><br>
        Owner: <input type="text" name="owner" value="<?= $row['owner'] ?>" required><br><br>
        <input type="submit" name="submit" value="Update Patient">
    </form>
    <br>
    <a href="index.php">Back</a>
</body>
</html>
